Medical History Listing

Subject-Level Medical History

Setup

See Prerequisites for installation instructions.

library(arframe)
library(pharmaverseadam)
library(dplyr, warn.conflicts = FALSE)

arm_levels <- c("Placebo", "Xanomeline Low Dose", "Xanomeline High Dose")

mh_listing <- pharmaverseadam::admh |>
  blank_to_na() |>
  filter(
    SAFFL == "Y",
    TRT01A %in% arm_levels,
    !is.na(MHBODSYS),
    !is.na(MHDECOD)
  ) |>
  mutate(TRT01A = factor(TRT01A, levels = arm_levels)) |>
  arrange(TRT01A, USUBJID, MHBODSYS, MHDECOD) |>
  select(
    TRT01A,
    USUBJID,
    MHBODSYS,
    MHDECOD,
    MHTERM,
    MHCAT,
    MHSEV
  )

arframe Pipeline

mh_listing |>
  fr_table() |>
  fr_titles(
    "Listing 16.2.4",
    "Medical History",
    "Safety Population"
  ) |>
  fr_cols(
    TRT01A  = fr_col(visible = FALSE),
    USUBJID = fr_col("Subject ID",       width = 1.2),
    MHBODSYS = fr_col("System Organ\nClass", width = 2.5),
    MHDECOD  = fr_col("Preferred\nTerm",     width = 2.0),
    MHTERM   = fr_col("Reported Term",   width = 2.0),
    MHCAT    = fr_col("Category",        width = 1.3),
    MHSEV    = fr_col("Severity",        width = 0.9, align = "center")
  ) |>
  fr_header(bold = TRUE, align = "center") |>
  fr_rows(page_by = "TRT01A", sort_by = c("USUBJID", "MHBODSYS", "MHDECOD"), suppress = "USUBJID") |>
  fr_footnotes(
    "Safety Population: all randomised subjects who received at least one dose of study medication.",
    "System Organ Class and Preferred Term are coded using MedDRA.",
    "Reported Term is the verbatim term as reported by the subject.",
    "Sorted by treatment arm, subject identifier, system organ class, and preferred term."
  )
